class C;
  bit [3:0] size;

  covergroup cov_size;
    coverpoint size;
  endgroup

  function new(input j = 0);    // Class constructor
    size = j;
    cov_size = new;             // Covergroup instance
  endfunction
endclass

  c3 = new c2;                  // Size = 10
  /*...*/

    c2.size = $urandom_range(0, 15);
    c2.cov_size.sample();
  end
  /*...*/
end
endmodule
