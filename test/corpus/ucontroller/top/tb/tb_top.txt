    logic        TXD;

        .Switches (Switches),
        .ROM_Data (ROM_Data),
        .ROM_Addr (ROM_Addr)
        );

    ucontroller I_UCONTROLLER (
        .Clk      (Clk),
        .Rst_n    (Rst_n),
        .TXD      (TXD),
        .RXD      (RXD),
        .Temp     (Temp),
        .Switches (Switches),
        .ROM_Data (ROM_Data),
        .ROM_Addr (ROM_Addr)
        );


endmodule
